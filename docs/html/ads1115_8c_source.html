93static ads1115_error_t write_register(ads1115_handle_t *handle, uint8_t reg_addr, uint16_t value)
94{
95 uint8_t data[2];
96 data[0] = (uint8_t)((value >> 8) & 0xFF);
97 data[1] = (uint8_t)(value & 0xFF);
98
99 return handle->i2c_write((uint8_t)handle->i2c_addr, reg_addr, data, 2) ? ADS1115_OK : ADS1115_ERROR_I2C_WRITE;
100}

109static ads1115_error_t read_register(ads1115_handle_t *handle, uint8_t reg_addr, uint16_t *value)
110{
111 uint8_t data[2];
112 if (!handle->i2c_read((uint8_t)handle->i2c_addr, reg_addr, data, 2))
113 {
114 return ADS1115_ERROR_I2C_READ;
115 }
116 *value = ((uint16_t)data[0] << 8) | data[1];
117 return ADS1115_OK;
118}

125static uint16_t build_config_register(const ads1115_config_t *config)
126{
127 uint16_t reg = 0;
128 reg |= ((uint16_t)config->mux << ADS1115_MUX_SHIFT) & ADS1115_MUX_MASK;
129 reg |= ((uint16_t)config->range << ADS1115_PGA_SHIFT) & ADS1115_PGA_MASK;
130 reg |= ((uint16_t)config->mode << ADS1115_MODE_SHIFT) & ADS1115_MODE_MASK;
131 reg |= ((uint16_t)config->data_rate << ADS1115_DR_SHIFT) & ADS1115_DR_MASK;
132 reg |= ((uint16_t)config->comp_mode << ADS1115_COMP_MODE_SHIFT) & ADS1115_COMP_MODE_MASK;
133 reg |= ((uint16_t)config->comp_pol << ADS1115_COMP_POL_SHIFT) & ADS1115_COMP_POL_MASK;
134 reg |= ((uint16_t)config->comp_latch << ADS1115_COMP_LAT_SHIFT) & ADS1115_COMP_LAT_MASK;
135 reg |= ((uint16_t)config->comp_queue << ADS1115_COMP_QUE_SHIFT) & ADS1115_COMP_QUE_MASK;
136 return reg;
137}

152static float raw_to_voltage(ads1115_range_t range, int16_t raw_value)
153{
154 if (range > ADS1115_RANGE_0V256)
155 range = ADS1115_RANGE_2V048;
156 return (raw_value * (float)ADS1115_FSR_VALUES[range]) / 32768.0f;
157}

167ads1115_error_t ads1115_init(ads1115_handle_t *handle)
168{
169 if (handle == NULL)
170 return ADS1115_ERROR_NULL_POINTER;
171 if (!handle->i2c_read || !handle->i2c_write || !handle->delay_ms)
172 return ADS1115_ERROR_INVALID_PARAM;
173
174 ads1115_error_t err;
175 if ((err = update_config_register(handle)) != ADS1115_OK)
176 return err;
177 if ((err = write_register(handle, ADS1115_REG_LO_THRESH, (uint16_t)handle->config.low_threshold)) != ADS1115_OK)
178 return err;
179 if ((err = write_register(handle, ADS1115_REG_HI_THRESH, (uint16_t)handle->config.high_threshold)) != ADS1115_OK)
180 return err;
181
182 handle->is_initialized = true;
183 return ADS1115_OK;
184}

205ads1115_error_t ads1115_get_range(ads1115_handle_t *handle, ads1115_range_t *range)
206{
207 if (!handle->is_initialized)
208 return ADS1115_ERROR_NOT_INITIALIZED;
209 if (range == NULL)
210 return ADS1115_ERROR_NULL_POINTER;
211 uint16_t reg;
212 ads1115_error_t err = read_register(handle, ADS1115_REG_CONFIG, &reg);
213 if (err == ADS1115_OK)
214 *range = (ads1115_range_t)((reg & ADS1115_PGA_MASK) >> ADS1115_PGA_SHIFT);
215 return err;
216}

246ads1115_error_t ads1115_continuous_conversion_read(ads1115_handle_t *handle, int16_t *adc_raw, float *voltage)
247{
248 if (!handle->is_initialized)
249 return ADS1115_ERROR_NOT_INITIALIZED;
250 if (!adc_raw || !voltage)
251 return ADS1115_ERROR_NULL_POINTER;
252 uint16_t raw;
253 ads1115_error_t err = read_register(handle, ADS1115_REG_CONVERSION, &raw);
254 if (err == ADS1115_OK)
255 {
256 *adc_raw = (int16_t)raw;
257 *voltage = raw_to_voltage(handle->config.range, *adc_raw);
258 }
259 return err;
260}

262ads1115_error_t ads1115_single_read(ads1115_handle_t *handle, int16_t *adc_raw, float *voltage)
263{
264 if (!handle->is_initialized)
265 return ADS1115_ERROR_NOT_INITIALIZED;
266 if (!adc_raw || !voltage)
267 return ADS1115_ERROR_NULL_POINTER;
268
269 uint16_t config_reg;
270 ads1115_error_t err = read_register(handle, ADS1115_REG_CONFIG, &config_reg);
271 if (err != ADS1115_OK)
272 return err;
273
274 config_reg |= ADS1115_OS_START_SINGLE;
275 if ((err = write_register(handle, ADS1115_REG_CONFIG, config_reg)) != ADS1115_OK)
276 return err;
277
278 /* Timing Calculation */
279 uint32_t wait_us = ADS1115_CONV_TIME_US[handle->config.data_rate];
280 handle->delay_ms((wait_us / 1000) + 1);
281
282 uint16_t raw;
283 if ((err = read_register(handle, ADS1115_REG_CONVERSION, &raw)) == ADS1115_OK)
284 {
285 *adc_raw = (int16_t)raw;
286 *voltage = raw_to_voltage(handle->config.range, *adc_raw);
287 }
288 return err;
289}

304}

306ads1115_error_t ads1115_is_ready(ads1115_handle_t *handle, bool *flag)
307{
308 if (!handle->is_initialized)
309 return ADS1115_ERROR_NOT_INITIALIZED;
310 if (!flag)
311 return ADS1115_ERROR_NULL_POINTER;
312 uint16_t reg;
313 ads1115_error_t err = read_register(handle, ADS1115_REG_CONFIG, &reg);
314 if (err == ADS1115_OK)
315 *flag = (reg & ADS1115_OS_MASK) != 0;
316 return err;
